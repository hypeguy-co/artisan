## Install

```bash
npm install --save artisan
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'artisan'
import 'artisan/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
